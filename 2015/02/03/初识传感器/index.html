初识传感器

发表于 2015-02-03   |   分类于 技术篇
  1. 加速计、晃动手势、陀螺仪的使用
  2. 加速计是用来检测手机收到加速度的大小和方向的，而手机静置的时候只受到重力加速度，所有又叫做重力感应器
#pragma mark 视图将要出现的时候设置加速度感应器
- (void)viewWillAppear:(BOOL)animated {
[super viewWillAppear:animated];
// 设置陀螺仪
UIAccelerometer *accelerometer = [UIAccelerometer sharedAccelerometer]; // 实例对象
accelerometer.delegate = self; // 设置代理为本身控制器
accelerometer.updateInterval = 0.1; // 设置传感器更新数据的时间间隔
NSLog(@"加速计设置成功");

[self.view becomeFirstResponder]; // 窗口对象作为第一响应者运动事件
}
#pragma mark - 重写----视图将要消失取消传感器
- (void)viewWillDisappear:(BOOL)animated {
[super viewWillDisappear:animated];
// 取消传感器，也就是设置代理为空
[[UIAccelerometer sharedAccelerometer] setDelegate:nil];
NSLog(@"加速计取消成功");
}
// 通过实现代理方法来检测手机方向的变化
#pragma mark - UIAccelerometerDelegate Methods
#pragma mark 获取加速计x/y/z轴上的改变
- (void)accelerometer:(UIAccelerometer *)accelerometer didAccelerate:(UIAcceleration *)acceleration {
NSLog(@"x = %f, y = %f, z = %f", acceleration.x, acceleration.y, acceleration.z);
}

// 用加速计侦测手机的晃动事件，需要在收到重力加速计的数据后做一些复杂的数学运算才能很好的实现，但是，UIResponder给我们提供了方法，已经帮我们完成了计算，我们只需使用下面的接口：
#pragma mark 重写----手机开始晃动时调用
- (void)motionBegan:(UIEventSubtype)motion withEvent:(UIEvent *)event {
NSLog(@"开始晃动");
}
#pragma mark 重写----晃动中断时调用
- (void)motionCancelled:(UIEventSubtype)motion withEvent:(UIEvent *)event {
NSLog(@"晃动中断时调用");
}
#pragma mark 重写----停止晃动时调用
- (void)motionEnded:(UIEventSubtype)motion withEvent:(UIEvent *)event {
NSLog(@"结束晃动");
}

// 由于窗口对象的第一响应者会收到所有的运动事件，为了确保应用当前视图控制器类为第一响应者，做如下操作：
#pragma mark 成为第一响应者
- (BOOL)canBecomeFirstResponder {
return YES;
}
#pragma mark 视图将要出现的时候设置加速度感应器
- (void)viewWillAppear:(BOOL)animated {
[super viewWillAppear:animated];
[self.view becomeFirstResponder]; // 窗口对象作为第一响应者运动事件
}

陀螺仪：陀螺仪内部有一个陀螺，它的轴由于陀螺效应始终与初始方向平行，这样就可以通过与初始方向的偏差计算出实际方向。iphone里陀螺仪实际上是一个结构精密的芯片，内部包含超微小的陀螺。
陀螺仪的强项在于测量设置的旋转运动。对设置自身的运动更擅长，所以适合需要精确方向控制的游戏。但不能确定设备方向。

#pragma mark - 重写----视图将要显示的时候设置陀螺仪
- (void)viewWillAppear:(BOOL)animated {
[super viewWillAppear:animated];
// 设置陀螺仪
CMMotionManager *motion = [[CMMotionManager alloc] init]; // 初始化对象
// 判断设备上的陀螺仪是否可用
if (motion.accelerometerAvailable) {
    // 可用，开始更新陀螺仪数据变化
    motion.accelerometerUpdateInterval = 0.01; // 更新频率
    [motion startGyroUpdates]; // 开始监听陀螺仪数据的变化
}
// 获取陀螺仪数据
double x = motion.gyroData.rotationRate.x;
double y = motion.gyroData.rotationRate.y;
double z = motion.gyroData.rotationRate.z;
NSLog(@"x = %f, y = %f, z = %f", x, y, z);
}
#pragma mark - 重写----视图将要消失的时候关闭陀螺仪数据更新
- (void)viewWillDisappear:(BOOL)animated {
[super viewWillDisappear:animated];
// 停止陀螺仪的数据更新
[[[CMMotionManager alloc] init] stopGyroUpdates];
}
